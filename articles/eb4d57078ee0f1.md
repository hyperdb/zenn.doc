def getIntMap():
    return map(int, input().split())


def getStringRow(N):
    return [list(input()) for _ in range(N)]


def main():
    h, w, x, y = getIntMap()
    s = getStringRow(h)

    n = 1
    for xp in range(x, h):
        if s[xp][y - 1] == '#':
            break
        n += 1
    for xp in range(x - 2, -1, -1):
        if s[xp][y - 1] == '#':
            break
        n += 1
    for yp in range(y, w):
        if s[x - 1][yp] == '#':
            break
        n += 1
    for yp in range(y - 2, -1, -1):
        if s[x - 1][yp] == '#':
            break
        n += 1

    print(n)


if __name__ == "__main__":
    main()
