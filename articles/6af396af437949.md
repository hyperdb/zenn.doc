def getInt():
    return int(input())


def getIntListRow(N):
    return [list(map(int, input().split())) for _ in range(N)]


def main():
    N = getInt()
    A = getIntListRow(2)

    max_sum = 0
    for i in range(N):
        # ä¸Šåˆ—ã®åˆè¨ˆã‚’è¨ˆç®—

        # ä¸‹åˆ—ã®åˆè¨ˆã‚’è¨ˆç®—

        # äº¤å·®éƒ¨åˆ†ã‚’è¨ˆç®—

    main()
