def getIntMap():
    return map(int, input().split())


def getString():
    return input()


def main():
    N, M = getIntMap()
    S = getString()
    T = getString()

    result = M * 10
    for i in range(N - M + 1):
        s = S[i : i + M]
        c = 0
        for j in range(M):
            x = int(s[j])
            y = int(T[j])
            c += x - y if x - y >= 0 else x - y + 10
        result = min(result, c)
    print(result)


if __name__ == "__main__":
    main()
