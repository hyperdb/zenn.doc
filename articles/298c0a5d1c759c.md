def getInt():
    return int(input())


def getString():
    return input()


def main():
    N = getInt()
    S = getString()

    ch = S[0]
    cnt = 1
    d = []
    for i in range(1, N):
        if S[i] != ch:
            d.append([ch, cnt])
            ch = S[i]
            cnt = 1
        else:
            cnt += 1
    d.append([ch, cnt])

    r = 1
    for i in range(1, len(d) - 1):
        # "/"ã€€ã‹ã¤ é€£ç¶šã—ã¦ã„ãªã„

                r = max(r, w * 2 + 1)

    print(r)


if __name__ == "__main__":
    main()
