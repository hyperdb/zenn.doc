def getIntMap():
    return map(int, input().split())


def getIntList():
    return list(map(int, input().split()))


def main():
    N, C = getIntMap()
    T = getIntList()

    # åˆå›žã¯è²°ãˆã‚‹
    t = T[0]
    r = 1

    for i in range(1, N):
        if T[i] - t >= C:
            t = T[i]
            r += 1
    print(r)


if __name__ == "__main__":
    main()
