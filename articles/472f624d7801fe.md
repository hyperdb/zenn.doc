## 提出結果

```python
def getIntList():
    return list(map(int, input().split()))


def getString():
    return input()


def main():
    RGB = getIntList()
    C = getString()

    cname = ["Red", "Green", "Blue"]
    x = cname.index(C)
    RGB[x] = 100

    print(min(RGB))


if __name__ == "__main__":
    main()

```
